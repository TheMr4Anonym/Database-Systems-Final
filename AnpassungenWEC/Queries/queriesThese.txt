1. Vergleich des Anteils erneuerbarer Energien und der Nettoenergieimporte pro Land
Query:

sql
Code kopieren
SELECT c.name, a.renewables_share_elec, e.net_elec_imports
FROM allrenewables a
JOIN energytrade e ON a.countryCode = e.countryCode AND a.year = e.year
JOIN country c ON a.countryCode = c.code
WHERE a.year = 2022
ORDER BY a.renewables_share_elec DESC;
2. CO₂-Emissionen in Ländern mit den höchsten Anteilen erneuerbarer Energien
Query:

sql
Code kopieren
SELECT c.name, a.renewables_share_elec, em.carbon_intensity_elec
FROM allrenewables a
JOIN emissions em ON a.countryCode = em.countryCode AND a.year = em.year
JOIN country c ON a.countryCode = c.code
WHERE a.year = 2022
ORDER BY a.renewables_share_elec DESC
LIMIT 10;
3. Durchschnittlicher Anteil erneuerbarer Energien und Nettoenergieimporte pro Kontinent
Query:

sql
Code kopieren
SELECT e.continent, AVG(a.renewables_share_elec) AS avg_renewables, AVG(et.net_elec_imports) AS avg_imports
FROM allrenewables a
JOIN energytrade et ON a.countryCode = et.countryCode AND a.year = et.year
JOIN encompasses e ON a.countryCode = e.country
WHERE a.year = 2022
GROUP BY e.continent;
4. Verteilung der Energieimporte basierend auf den Hauptquellen erneuerbarer Energie (Solar, Wind, Hydro)
Query:

sql
Code kopieren
SELECT c.name, s.solar_share_elec, w.wind_share_elec, h.hydro_share_elec, e.net_elec_imports
FROM solarenergy s
JOIN windenergy w ON s.countryCode = w.countryCode AND s.year = w.year
JOIN hydroenergy h ON s.countryCode = h.countryCode AND s.year = h.year
JOIN energytrade e ON s.countryCode = e.countryCode AND s.year = e.year
JOIN country c ON s.countryCode = c.code
WHERE s.year = 2022;
5. Trendanalyse: Wachstum erneuerbarer Energien und Rückgang der CO₂-Emissionen (2010–2022)
Query:

sql
Code kopieren
SELECT a.year, AVG(a.renewables_share_elec) AS avg_renewables, AVG(em.carbon_intensity_elec) AS avg_carbon
FROM allrenewables a
JOIN emissions em ON a.countryCode = em.countryCode AND a.year = em.year
WHERE a.year BETWEEN 2010 AND 2022
GROUP BY a.year
ORDER BY a.year;
6. Länder mit überdurchschnittlichem Anteil erneuerbarer Energien und ihren Nettoenergieimporten
Query:

sql
Code kopieren
WITH avg_renewables AS (
    SELECT AVG(renewables_share_elec) AS avg_renewables
    FROM allrenewables
    WHERE year = 2022
)
SELECT c.name, a.renewables_share_elec, e.net_elec_imports
FROM allrenewables a
JOIN energytrade e ON a.countryCode = e.countryCode AND a.year = e.year
JOIN country c ON a.countryCode = c.code
WHERE a.year = 2022 AND a.renewables_share_elec > (SELECT avg_renewables FROM avg_renewables)
ORDER BY a.renewables_share_elec DESC;

8. Zusammenhang zwischen Energieverbrauch pro Kopf und Anteil erneuerbarer Energien
Query:

sql
Code kopieren
SELECT c.name, f.energy_per_capita, a.renewables_share_elec
FROM efficiency f
JOIN allrenewables a ON f.countryCode = a.countryCode AND f.year = a.year
JOIN country c ON f.countryCode = c.code

9. Länder mit minimalen Energieimporten und ihren Hauptquellen erneuerbarer Energie
Query:

sql
Code kopieren
SELECT c.name, e.net_elec_imports, MAX(s.solar_share_elec) AS solar, MAX(w.wind_share_elec) AS wind, MAX(h.hydro_share_elec) AS hydro
FROM energytrade e
JOIN solarenergy s ON e.countryCode = s.countryCode AND e.year = s.year
JOIN windenergy w ON e.countryCode = w.countryCode AND e.year = w.year
JOIN hydroenergy h ON e.countryCode = h.countryCode AND e.year = h.year
JOIN country c ON e.countryCode = c.code
WHERE e.year = 2022
GROUP BY c.name, e.net_elec_imports
ORDER BY e.net_elec_imports ASC
LIMIT 10;
